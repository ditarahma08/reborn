import PrivateLayout from '@components/layouts/PrivateLayout';
import { Button, Card, CardSubtitle, CardTitle, Container } from '@components/otoklix-elements';
import { useAuth } from '@contexts/auth';
import { api, authenticateAPI } from '@utils/API';
import { sentryBreadcrumb } from '@utils/globalFunctions';
import { gtag } from '@utils/Gtag';
import Helper from '@utils/Helper';
import moment from 'moment';
import { useRouter } from 'next/router';
import { useEffect, useState } from 'react';
 
sentryBreadcrumb('pages/thank-you/index');
 
const PaymentStatus = () => {
  const router = useRouter();
  const { booking_code } = router.query;
  const { source } = router.query;
  let gTagLabel = '';
  if (source == 'landing') gTagLabel = 'landing-page';
  const { token } = useAuth();
 
  const [data, setData] = useState({});
  const [paymentButtonLoading, setPaymentButtonLoading] = useState(false);
  const [paymentLink, setPaymentLink] = useState(false);
 
  const fetchPaymentStatus = async () => {
    authenticateAPI(token);
 
    const response = await api.get(`v2/bookings/${booking_code}/`);
    return response.data.data;
  };
 
  const prepareInitialData = async () => {
    const respData = await fetchPaymentStatus();
    setData(respData);
  };
 
  const preparePaymentLink = async () => {
    const respData = await fetchPaymentStatus();
    if (respData?.midtrans_link) {
      setPaymentButtonLoading(false);
      setPaymentLink(respData?.midtrans_link);
    }
  };
 
  useEffect(() => {
    const interval = setInterval(() => {
      if (paymentButtonLoading) {
        preparePaymentLink();
      }
    }, 3000);
 
    return () => clearInterval(interval);
  });
 
  useEffect(() => {
    if (booking_code) {
      prepareInitialData();
      gtag('view thank you page', 'viewThankYouPage', gTagLabel);
    }
  }, [booking_code]);
 
  useEffect(() => {
    const paymentUrl = data?.midtrans_link;
    const workshopTier = data?.workshop?.tier;
    if (workshopTier && workshopTier?.value !== 'tier_2') {
      if (paymentUrl) {
        setPaymentLink(paymentUrl);
      } else {
        setPaymentButtonLoading(true);
      }
    }
  }, [data]);
 
  const goToPaymentLink = () => {
    router.push(paymentLink);
  };
  const goToWorkshop = () => {
    if (data?.workshop?.slug) {
      router.push(`/bengkel/${data?.workshop?.slug}/`);
    }
  };
  const showPaymentButton = (paymentLink || paymentButtonLoading) && data?.total_price !== 0;
  const showPromoSection = data?.promo && data?.promo?.is_campaign == 0;
 
  return (
    <PrivateLayout hasAppBar={false} wrapperClassName="wrapper-full bg-bg">
      <Container className="pt-3 pb-4">
        <Card className="px-3 py-4 mt-1 mb-2 border-0 payment-status bg-off-white">
          <Card className="p-3 mb-3 fw-bold text-center border-0 payment-status-card">
            <div className="d-flex justify-content-center align-items-center">
              <img src="/assets/icons/success.svg" alt="" />
              <span className="ms-2 text-success">Booking Berhasil</span>
            </div>
 
            <span className="mt-3 fs-4">{`Rp${Helper.formatMoney(data.total_price)}`}</span>
          </Card>
 
          <div className="mt-1 text-center">
            <span className="fw-bold fs-6 text-title-active">Ringkasan Pesanan</span>
          </div>
 
          <div className="mt-3 mb-1 d-flex justify-content-between">
            <span className="left-text">Order ID</span>
            <span className="right-text">{data.booking_code}</span>
          </div>
 
          <div className="mb-1 d-flex justify-content-between">
            <span className="left-text">Mobil</span>
            <span className="right-text">{`${data?.user_car?.car_details?.car_model?.brand?.name} ${data?.user_car?.car_details?.car_model?.model_name}`}</span>
          </div>
 
          <div className="mb-1 d-flex">
            <div className="me-auto w-50">
              <span className="left-text w-100">Bengkel</span>
            </div>
            <div className="text-end">
              <a
                href={data?.workshop?.gmaps_link}
                className="open-map-link pointer text-primary"
                target="_blank"
                rel="noreferrer">
                <span className="right-text">{data.workshop?.name}</span>
                <img className="ms-1" src="/assets/icons/open-new-link.svg" alt="" />
              </a>
            </div>
          </div>
 
          <hr className="my-3" />
 
          <span className="mb-1 section-title">Jadwal Booking</span>
 
          <div className="mb-1 d-flex justify-content-between">
            <span className="left-text">Tanggal</span>
            <span className="right-text">{moment(data.booking_datetime).format('D MMM YYYY')}</span>
          </div>
 
          <div className="mb-1 d-flex justify-content-between">
            <span className="left-text">Jam</span>
            <span className="right-text">{`${moment(data.booking_datetime).format(
              'HH:mm'
            )} WIB`}</span>
          </div>
 
          {data?.maximum_confirmation_time && (
            <Card className="card-common-info p-2 mt-3">
              <CardTitle className="mb-3 text-success">
                <img src="/assets/icons/exclamation-green.svg" /> Info
              </CardTitle>
              <CardSubtitle className="text-label">
                Kami akan segera mengkonfirmasi ketersediaan servis kamu di bengkel ini lewat
                notifikasi dan pesan WhatsApp maksimal pada{' '}
                <span className="fw-bold">
                  {moment(data?.maximum_confirmation_time).format('D MMM YYYY')}
                  {', jam '}
                  {moment(data?.maximum_confirmation_time).format('HH:mm')}{' '}
                </span>
                <span role="img" aria-label="moce">
                  ðŸ˜‰
                </span>
              </CardSubtitle>
            </Card>
          )}
 
          {showPaymentButton && (
            <Button
              onClick={goToPaymentLink}
              className="mt-3"
              size="sm"
              color="secondary"
              disabled={paymentButtonLoading}
              loading={paymentButtonLoading}>
              {' '}
              Bayar
            </Button>
          )}
 
          <hr className="my-3" />
 
          <span className="mb-1 section-title">Rincian Paket / Servis</span>
 
          {data?.packages?.length > 0 &&
            data.packages.map((packageItem) => {
              return (
                <div key={packageItem?.name} className="package-detail">
                  <div className="package-name mt-1 mb-2">
                    {Helper.shortenName(packageItem?.name)}
                  </div>
 
                  {packageItem.package_details.map((item, index) => {
                    let showOriginalPrice = false;
                    if (item?.original_price > item?.price) {
                      showOriginalPrice = true;
                    }
                    return (
                      <div key={index} className="package-item-box text-placeholder mb-3">
                        <div>
                          <div className="fs-7 fw-semi-bold ">{item.name}</div>
                          <div className="subtitle">{item?.line_item?.items?.name}</div>
                        </div>
                        <div className="text-end">
                          {showOriginalPrice && (
                            <div className="subtitle text-decoration-line-through">{`Rp${Helper.formatMoney(
                              item?.original_price
                            )}`}</div>
                          )}
                          <div className="fs-7 fw-semi-bold text-dark">{`Rp${Helper.formatMoney(
                            item?.price
                          )}`}</div>
                        </div>
                      </div>
                    );
                  })}
                </div>
              );
            })}
 
          <hr className="my-3" />
 
          {data?.total_point_spent > 0 && (
            <div className="mb-2 d-flex justify-content-between">
              <span className="left-text">OtoPoints</span>
              <span className="right-text">
                {data?.total_point_spent > 0 ? `-${Helper.formatMoney(data.otopoints_used)}` : '0'}
              </span>
            </div>
          )}
 
          {showPromoSection && (
            <div className="mb-2 d-flex justify-content-between align-items-center">
              <div>
                <div className="left-text">Promo</div>
                <div className="subtitle fw-medium text-light">{data.promo.promo_code}</div>
              </div>
              <span className="right-text">{`-${Helper.formatMoney(data.discount_promo)}`}</span>
            </div>
          )}
 
          <div className="mb-2 d-flex justify-content-between text-success total">
            <span className="text-left">Total Bayar</span>
            <span className="text-right">{`Rp${Helper.formatMoney(data.total_price)}`}</span>
          </div>
 
          {data?.total_point_get > 0 && (
            <Card className="card-info-otopoints p-2">
              <CardTitle className="mb-2">Total OtoPoints akan didapatkan</CardTitle>
              <CardSubtitle className="text-primary">{`+ ${Helper.formatMoney(
                data?.total_point_get
              )} Points`}</CardSubtitle>
            </Card>
          )}
        </Card>
 
        <Button className="mt-4" block size="sm" color="secondary" onClick={goToWorkshop}>
          Booking Lagi
        </Button>
 
        <Button
          className="mt-3"
          block
          size="sm"
          outline
          color="secondary"
          onClick={() => router.push(`/`)}>
          Ke Home
        </Button>
      </Container>
    </PrivateLayout>
  );
};
 
export default PaymentStatus;
